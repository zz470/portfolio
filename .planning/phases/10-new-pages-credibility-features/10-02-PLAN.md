       export function buildFAQSchema(entries: readonly { question: string; answer: string }[]) {
         return {
           "@context": "https://schema.org",
           "@type": "FAQPage",
           "mainEntity": entries.map((entry) => ({
             "@type": "Question",
             "name": entry.question,
             "acceptedAnswer": {
               "@type": "Answer",
               "text": entry.answer,
             },
           })),
         };
       }
